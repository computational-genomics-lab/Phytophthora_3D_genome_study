##significantly different PFAM domain


library(tidyverse)
library(purrr)


data <- read.table("genes_with_pfam_compartment_AB.bed", 
                   header = FALSE, sep = "\t", stringsAsFactors = FALSE)

# Rename columns for clarity
colnames(data) <- c("Scaffold", "Start", "End", "Gene", "Domain", "Compartment")

# Count domain frequencies in each compartment
domain_counts <- data %>% 
    group_by(Domain, Compartment) %>% 
    summarise(Frequency = n(), .groups = "drop")

# Reshape data for statistical comparison
domain_wide <- domain_counts %>%
    pivot_wider(names_from = Compartment, values_from = Frequency, values_fill = list(Frequency = 0))

# Perform Fisher's exact test for each domain
domain_wide <- domain_wide %>%
    mutate(
        Total_A = sum(domain_wide$A),  # Total genes in A
        Total_B = sum(domain_wide$B),  # Total genes in B
        P_value = map2_dbl(A, B, ~fisher.test(matrix(c(.x, sum(domain_wide$A) - .x, 
                                                       .y, sum(domain_wide$B) - .y), 
                                                     nrow = 2))$p.value)
    )

# Save full statistics (with all P-values)
write.table(domain_wide, file = "all_domains_with_pvalues.txt", sep = "\t", row.names = FALSE)

# Filter for significant domains (P-value < 0.05)
significant_domains <- domain_wide %>%
    filter(P_value < 0.05)

# Save only significant results
write.table(significant_domains, file = "significant_domains_with_pvalues.txt", sep = "\t", row.names = FALSE)

# Merge significant domains with original data for plotting
plot_data <- domain_counts %>%
    filter(Domain %in% significant_domains$Domain)

# Create the bubble plot
ggplot(plot_data, aes(x = Compartment, y = Domain, size = Frequency, color = Compartment)) +
    geom_point(alpha = 0.7) +
    scale_size(range = c(3, 10)) +  # Adjust bubble size
    scale_color_manual(values = c("A" = "hotpink1", "B" = "skyblue"), name = "Compartment") +
    labs(title = "Significantly Different Domains by Compartment",
         x = "Compartment", y = "Domain", size = "Frequency") +
    theme_minimal() +
    theme(
        axis.text.x = element_text(angle = 90, hjust = 1, face = "bold"),
        axis.text.y = element_text(face = "bold"),
        axis.title.x = element_text(face = "bold"),
        axis.title.y = element_text(face = "bold"),
        plot.title = element_text(face = "bold", hjust = 0.5),
        legend.title = element_text(face = "bold")
    )
