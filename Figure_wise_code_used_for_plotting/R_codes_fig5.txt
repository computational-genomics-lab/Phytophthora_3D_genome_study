##Gene densityAB


# Load required library
library(ggplot2)

# Load gene density data for compartments A and B
compartment_A_gene <- read.table("final_A_gene_density.bedgraph", header = FALSE)
compartment_B_gene <- read.table("final_B_gene_density.bedgraph", header = FALSE)

# Extract gene density (5th column)
gene_density_A <- compartment_A_gene$V5
gene_density_B <- compartment_B_gene$V5

# Create a combined data frame
gene_data <- data.frame(
    Compartment = c(rep("A", length(gene_density_A)), rep("B", length(gene_density_B))),
    Gene_Density = c(gene_density_A, gene_density_B)
)

# Perform Welch's t-test
t_test_result_gene <- t.test(Gene_Density ~ Compartment, data = gene_data)

# Determine significance label based on p-value
significance_label_gene <- ifelse(t_test_result_gene$p.value < 0.0001, "****",
                                  ifelse(t_test_result_gene$p.value < 0.001, "***",
                                         ifelse(t_test_result_gene$p.value < 0.01, "**",
                                                ifelse(t_test_result_gene$p.value < 0.05, "*", "ns"))))

# Print the t-test result
print(t_test_result_gene)


ggplot(gene_data, aes(x = Compartment, y = Gene_Density, fill = Compartment)) +
    # Violin plot
    geom_violin(trim = FALSE, alpha = 0.6, color = NA) +

    # Boxplot with colored fill from aes
    geom_boxplot(outlier.shape = 16, outlier.colour = "red", 
                 width = 0.2, color = "black") +

    labs(title = "Comparison of Gene Density Between A and B Compartments",
         y = "Gene Density (per 10kb)",
         x = "Compartment") +

    theme_minimal(base_size = 15) +

    # Significance stars
    annotate("text", x = 1.5, y = max(gene_data$Gene_Density) * 1.05, 
             label = significance_label_gene, size = 6, color = "black", fontface = "bold") +

    # p-value text
    annotate("text", x = 1.5, y = max(gene_data$Gene_Density) * 1.1, 
             label = paste0("p = ", formatC(t_test_result_gene$p.value, format = "e", digits = 2)), 
             size = 5, color = "black", fontface = "italic") +

    theme(axis.title = element_text(size = 14, face = "bold"),
          axis.text = element_text(size = 12),
          plot.title = element_text(size = 16, face = "bold", hjust = 0.5),
          legend.position = "none") +

    scale_fill_manual(values = c("A" = "#FF6347", "B" = "#4682B4"))

# Assign column names
colnames(A_tRNA) <- c("Scaffold", "Start", "End", "Density", "tRNA_Count")
colnames(B_tRNA) <- c("Scaffold", "Start", "End", "Density", "tRNA_Count")

# Add compartment labels
A_tRNA$Compartment <- "A"
B_tRNA$Compartment <- "B"

# Combine datasets
data <- rbind(A_tRNA, B_tRNA)

# Wilcoxon test
wilcox_test <- wilcox.test(tRNA_Count ~ Compartment, data = data)
p_value <- wilcox_test$p.value  # Extract p-value

# Calculate mean tRNA count per 10kb for each compartment
mean_counts <- data %>%
  group_by(Compartment) %>%
  summarise(mean_tRNA = mean(tRNA_Count, na.rm = TRUE))

# Print Wilcoxon test result
print(wilcox_test)

# Create jitter plot with mean and p-value
ggplot(data, aes(x = Compartment, y = tRNA_Count, color = Compartment)) +
  geom_jitter(width = 0.2, alpha = 0.6, size = 1.5) +  # Jitter points
  stat_summary(fun = mean, geom = "point", shape = 23, size = 2.5, fill = "black") +  # Mean points
  annotate("text", x = 1.5, y = max(data$tRNA_Count, na.rm = TRUE) + 1,
           label = paste("Wilcoxon p =", signif(p_value, digits = 3)), size = 4) +  # Add p-value
  annotate("text", x = 1, y = max(data$tRNA_Count, na.rm = TRUE) + 2,
           label = paste("Mean A =", round(mean_counts$mean_tRNA[mean_counts$Compartment == "A"], 3)), size = 4) +  # Mean for A
  annotate("text", x = 2, y = max(data$tRNA_Count, na.rm = TRUE) + 2,

  scale_color_manual(values = c("A" = "#FF6347", "B" = "#4682B4")) +  # Custom colors
  labs(
    title = "Comparison of tRNA Counts in A and B Compartments",
    x = "Compartment",
    y = "tRNA Count (per 10kb)"
  ) +
  theme_minimal() +
  theme(
    axis.text.x = element_text(size = 12),

    axis.title.x = element_text(size = 12),
    axis.title.y = element_text(size = 12),

library(tidyverse)

# Read data
effector_domain_data <- read.table("all_effectors_curated_hmm_AB.bed", 
                                   sep = "\t", header = FALSE, stringsAsFactors = FALSE)

colnames(effector_domain_data) <- c("Scaffold", "Start", "End", "Domain", "Compartment")

# Calculate frequency of each domain in compartments A and B
domain_frequencies <- effector_domain_data %>% 
    group_by(Domain, Compartment) %>%
    summarise(Frequency = n(), .groups = "drop")

# Get total frequency of each domain
top_10_domains <- domain_frequencies %>%
    group_by(Domain) %>%
    summarise(Total_Frequency = sum(Frequency)) %>%
    top_n(10, Total_Frequency)

# Filter for top 10 domains
top_10_domain_frequencies <- domain_frequencies %>%
    filter(Domain %in% top_10_domains$Domain)

# Define compartment sizes (MB)
compartment_sizes <- c(A = 41.12, B = 35.58)  # MB

# Add density column
top_10_domain_frequencies <- top_10_domain_frequencies %>%
    mutate(Density_per_MB = Frequency / compartment_sizes[Compartment])

# Pivot for saving
number_table <- top_10_domain_frequencies %>%
    pivot_wider(names_from = Compartment, values_from = Frequency, values_fill = 0) %>%
    rename(Number_A = A, Number_B = B)

    pivot_wider(names_from = Compartment, values_from = Density_per_MB, values_fill = 0) %>%

  mutate(Chromosome = factor(Chromosome, levels = paste0("Chromosome_", 1:17)))
