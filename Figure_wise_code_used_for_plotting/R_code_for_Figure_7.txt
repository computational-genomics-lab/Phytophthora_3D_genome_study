find_overlap <- function(expr_df, regions_df, label) {
  regions_df <- regions_df %>% rename(start_region = start, end_region = end)
  inner_join(expr_df, regions_df, by = "chr") %>%
    filter(start >= start_region & end <= end_region) %>%
    mutate(region_type = label) %>%
    select(-start_region, -end_region)
}

gene_distances <- distanceToNearest(genes_gr, boundary_gr_all)

  gene_id = genes_gr$gene_id[queryHits(gene_distances)],
  distance = mcols(gene_distances)$distance

# 📊 Plotting function
plot_expr_distance <- function(condition, label) {
  expr_df <- expr %>%
    mutate(
      log2_TPM = log2(!!sym(condition) + 1),
      gene_id = rownames(.)
    )

  merged_df <- left_join(expr_df, gene_distance_df, by = "gene_id") %>%
    filter(!is.na(distance))

  merged_df$distance_group <- cut(
    merged_df$distance,
    breaks = c(-1, 1, 10000, 20000, 30000, 40000, Inf),
    labels = c("0 (Boundary)", "<10kb", "10-20kb", "20-30kb", "30-40kb", ">40kb")
  )

  present_groups <- levels(droplevels(merged_df$distance_group))
  all_comparisons <- list(
    c("0 (Boundary)", "<10kb"),
    c("0 (Boundary)", "10-20kb"),
    c("0 (Boundary)", "20-30kb"),
    c("0 (Boundary)", "30-40kb"),
    c("0 (Boundary)", ">40kb")
  )
  valid_comparisons <- Filter(function(comp) all(comp %in% present_groups), all_comparisons)

  p <- ggplot(merged_df, aes(x = distance_group, y = log2_TPM, fill = distance_group)) +
    geom_boxplot(outlier.shape = 21, outlier.size = 1.2, alpha = 0.9) +
    stat_summary(fun = mean, geom = "point", shape = 23, size = 3, fill = "white") +
    stat_compare_means(method = "wilcox.test", comparisons = valid_comparisons, 
                       method.args = list(alternative = "greater"), label = "p.signif") +
    labs(
      title = label,
      x = "Distance to Nearest Boundary",
      y = "log2(TPM + 1)"
    ) +
    theme_minimal(base_size = 14) +
    theme(
      plot.title = element_text(hjust = 0.5, face = "bold", size = 12),
      axis.text.x = element_text(angle = 30, hjust = 1),
      legend.position = "none"
    )
  return(p)
}

# 🔁 Generate plots for all 6 conditions
